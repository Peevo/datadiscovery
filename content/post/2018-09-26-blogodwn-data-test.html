A quick test of loading in new data and sending to blog.

colnames(teach.salaries) <- c("Location", "Indicator", "Subject", "Measure", "Frequency", "Time", "Value", "Flag Codes")
x <- teach.salaries %>% group_by() %>% filter(Subject == "PRY_START", Location == "AUS" | Location == "USA") %>% select(Location, Time, Value)
ggplot(x, aes(Time, Value)) + geom_point(aes(colour = factor(Location)))
